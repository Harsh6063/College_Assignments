import random
import string

def generate_random_key():
    # Note: 'J' is typically omitted or combined with 'I' in Playfair
    alphabet = list("ABCDEFGHIKLMNOPQRSTUVWXYZ")
    random.shuffle(alphabet)
    return "".join(alphabet)

def generate_matrix(key):
    matrix = [list(key[i:i+5]) for i in range(0, 25, 5)]
    return matrix

def find_position(matrix, char):
    for i in range(5):
        for j in range(5):
            if matrix[i][j] == char:
                return i, j
    return None

def prepare_text(text):
    text = text.upper().replace("J", "I")
    text = "".join(filter(str.isalpha, text))
    pairs = []
    i = 0
    while i < len(text):
        a = text[i]
        if i+1 < len(text):
            b = text[i+1]
            if a == b:
                pairs.append(a + 'X')
                i += 1
            else:
                pairs.append(a + b)
                i += 2
        else:
            pairs.append(a + 'X')
            i += 1
    return pairs

def encrypt_playfair(text, matrix):
    pairs = prepare_text(text)
    result = ""
    for pair in pairs:
        r1, c1 = find_position(matrix, pair[0])
        r2, c2 = find_position(matrix, pair[1])
        
        if r1 == r2: # Same row: shift right
            result += matrix[r1][(c1 + 1) % 5]
            result += matrix[r2][(c2 + 1) % 5]
        elif c1 == c2: # Same column: shift down
            result += matrix[(r1 + 1) % 5][c1]
            result += matrix[(r2 + 1) % 5][c2]
        else: # Rectangle rule: swap columns
            result += matrix[r1][c2]
            result += matrix[r2][c1]
    return result

# --- Execution ---
key = generate_random_key()
matrix = generate_matrix(key)

print("Generated Key Matrix:")
for row in matrix:
    print(" ".join(row))

plaintext = input("\nEnter plaintext: ")
cipher = encrypt_playfair(plaintext, matrix)
print("Ciphertext:", cipher)
